  SELECT email_id
       , CAST(player_id AS INT64) AS player_id
       , 1 AS voted_for_player
       , SUM(total_votes) AS votes_for_player
    FROM data_science_sbx.asb_2019_votes
   WHERE player_id NOT IN('May', 'January')
   GROUP BY 1, 2

), sales_for_player AS (

      SELECT sh.email_id
           , pid.player_id
           , 1 AS purchased_for_player
           , COUNT(*) AS purchases_for_player
      FROM customer_sales.mlb_shop_product_master AS products
      JOIN customer_sales.team_master_mlb                    AS tm ON products.team_id = TM.team_id
      JOIN customer_sales.mlb_shop_product_sales_combined_vw AS sh ON sh.sku = products.sku
      JOIN data_science_sbx.player_avidity_player_skus       AS pid ON pid.sku = products.sku
     WHERE pid.player_id IS NOT NULL
       AND CAST(order_date AS DATE) >= DATE '2016-01-01'
     GROUP BY sh.email_id, pid.player_id

), joint_fans AS (
  SELECT email_id FROM votes_for_player GROUP BY 1

  UNION ALL

  SELECT email_id FROM sales_for_player GROUP BY 1
  
), unique_fans AS (

  SELECT email_id FROM joint_fans LIMIT 1000

), active_players AS (

  SELECT id AS player_id
       , nameFirstLast AS player_name
       , name_abbrev AS team_code
       , pop.popularity
       , p.primaryPosition.abbreviation AS position

    JOIN teradata_exports.team_master_mlb AS t ON p.currentTeam.id = t.team_id_other

   WHERE season = 2019
     AND week = 27

  SELECT email_id, player_id, player_name, team_code, position, popularity
    FROM unique_fans
   CROSS JOIN active_players

), total_votes AS (

def sigmoid(z):
    return 1.0 / (1.0 + np.exp(-z))

def learning_schedule(t):
    return 5. / (t + 10.)

def update_theta_p(theta_p, epoch, i):
    random_index = np.random.randint(len(df), size=batch_size)
    #
    x = df.iloc[random_index][sales_features].values
    y = df.iloc[random_index]['purchased_for_player']
    #
    mu       = np.dot(theta_p.T, x.T)
    h        = sigmoid(mu)
    gradient = np.dot(x.T, (h - y)) / y.shape[0]
    eta      = learning_schedule(epoch * m + i)
    theta_p_ = theta_p + eta * gradient
    return theta_p_

def update_theta_v(theta_v, epoch, i):
    random_index = np.random.randint(len(df), size=batch_size)
    #
    x = df.iloc[random_index][votes_features].values
    y = df.iloc[random_index]['voted_for_player']
    #
    mu       = np.dot(theta_v.T, x.T)
    h        = sigmoid(mu)
    gradient = np.dot(x.T, (h - y)) / y.shape[0]
    eta      = learning_schedule(epoch * m + i)
    theta_v_ = theta_v + eta * gradient
    return theta_v_

def update_z(df, theta_p, theta_v, w, epoch, i):
    s       = df[causal_features].values
    t_u_b_z = np.dot(s, w)
    y_z     = df['z']
    pt_0    = (y_z - sigmoid(t_u_b_z)) *w[0]
    pt_1    = (y_z - sigmoid(t_u_b_z)) *w[1]
    #pt_0    = w[0]*(sigmoid(t_u_b_z) - y_z)
    #pt_1    = w[1]*(sigmoid(t_u_b_z) - y_z)
    #
    x_p     = df[sales_features].values
    y_p     = df['purchased_for_player']
    mu_p    = np.dot(theta_p.T, x_p.T)
    h_p     = sigmoid(mu_p)
    pt_p    = theta_p[-1]*(h_p - y_p) #/ y_p.shape[0]
    #
    x_v     = df[votes_features].values
    y_v     = df['voted_for_player']
    mu_v    = np.dot(theta_v.T, x_v.T)
    h_v     = sigmoid(mu_v)
    pt_v    = theta_v[-1]*(h_v - y_v) #/ y_p.shape[0]
    #
    z_gradient = pt_0 + pt_1 + pt_p + pt_v
    new_z      = y_z - z_gradient * learning_schedule(i)
    df['old_z'] = df['z']
    df['z'] = new_z
    df['pt_0'] = pt_0
    df['pt_1'] = pt_1
    df['pt_p'] = pt_p
    df['pt_v'] = pt_v
    df['z_diff'] = df['z'] - df['old_z']
    return df

def update_w():
    random_index = np.random.randint(len(df), size=batch_size)
    #
    s = df.iloc[random_index][causal_features].values
    y = df.iloc[random_index]['z']
    mu = np.dot(s, w)
    h   = sigmoid(mu)
    gradient = np.dot(s.T, (h - y)) / y.shape[0]
    eta      = learning_schedule(epoch * m + i)
    w_  = w + eta * gradient
    return w_

    neither_txn = df.loc[(df['voted_for_player'] == 0) & (df['purchased_for_player'] == 0)]

    print("------------------------------------")

query = """
WITH votes_for_player AS (

  SELECT email_id
       , CAST(player_id AS INT64) AS player_id
       , 1 AS voted_for_player
       , SUM(total_votes) AS votes_for_player
    FROM data_science_sbx.asb_2019_votes
   WHERE player_id NOT IN('May', 'January')
   GROUP BY 1, 2

), sales_for_player AS (

      SELECT sh.email_id
           , pid.player_id
           , 1 AS purchased_for_player
           , COUNT(*) AS purchases_for_player
      FROM customer_sales.mlb_shop_product_master AS products
      JOIN customer_sales.team_master_mlb                    AS tm ON products.team_id = TM.team_id
      JOIN customer_sales.mlb_shop_product_sales_combined_vw AS sh ON sh.sku = products.sku
      JOIN data_science_sbx.player_avidity_player_skus       AS pid ON pid.sku = products.sku
     WHERE pid.player_id IS NOT NULL
       AND CAST(order_date AS DATE) >= DATE '2016-01-01'
     GROUP BY sh.email_id, pid.player_id

), joint_fans AS (
  SELECT email_id FROM votes_for_player GROUP BY 1

  UNION ALL

  SELECT email_id FROM sales_for_player GROUP BY 1
  
), unique_fans AS (

  SELECT email_id FROM joint_fans WHERE email_id = 99671371

), active_players AS (

  SELECT id AS player_id
       , nameFirstLast AS player_name
       , name_abbrev AS team_code
       , pop.popularity
       , p.primaryPosition.abbreviation AS position
    FROM customer_sales_db.stats_api_mlb_players_current_year AS p
    JOIN teradata_exports.team_master_mlb AS t ON p.currentTeam.id = t.team_id_other
    JOIN data_science_sbx.player_avidity_overall_popularity AS pop ON p.id = pop.player_id
   WHERE season = 2019
     AND week = 27
     AND pop.popularity IS NOT NULL

), fans_players AS (

  SELECT email_id, player_id, player_name, team_code, position, popularity
    FROM unique_fans
   CROSS JOIN active_players

), total_votes AS (

  SELECT email_id, SUM(total_votes) AS total_votes FROM data_science_sbx.asb_2019_votes WHERE player_id NOT IN('May', 'January') GROUP BY 1

), total_sales AS (

      SELECT sh.email_id
           , COUNT(DISTINCT sh.order_number) AS total_sales
      FROM customer_sales.mlb_shop_product_master AS products
      JOIN customer_sales.team_master_mlb                    AS tm ON products.team_id = TM.team_id
      JOIN customer_sales.mlb_shop_product_sales_combined_vw AS sh ON sh.sku = products.sku
      JOIN data_science_sbx.player_avidity_player_skus       AS pid ON pid.sku = products.sku
     WHERE pid.player_name IS NOT NULL
       AND CAST(order_date AS DATE) >= DATE '2016-01-01'
     GROUP BY sh.email_id

)
SELECT fans_players.email_id
     , fans_players.player_id
     , fans_players.player_name
     , fans_players.team_code
     , fans_players.position
     --
     , COALESCE(v.voted_for_player, 0) AS voted_for_player
     , COALESCE(v.votes_for_player, 0) AS votes_for_player
     , COALESCE(total_votes.total_votes, 0) AS total_votes
     --
     , COALESCE(purchased_for_player, 0) AS purchased_for_player
     , COALESCE(purchases_for_player, 0) AS purchases_for_player
     , COALESCE(total_sales.total_sales, 0) AS total_sales
     --
     , COALESCE(lifetime_avidity_scores.attention, 0.0) AS attention
     , COALESCE(lifetime_avidity_scores.communication, 0.0) AS explicit
     , COALESCE(lifetime_avidity_scores.spend, 0.0) AS spend
     , COALESCE(lifetime_avidity_scores.avidity, 0.0) AS avidity
     --
     , fans_players.popularity
  FROM fans_players
  LEFT JOIN votes_for_player AS v ON fans_players.email_id = v.email_id AND fans_players.player_id = v.player_id
  LEFT JOIN total_votes      ON fans_players.email_id = total_votes.email_id
  LEFT JOIN sales_for_player ON fans_players.email_id = sales_for_player.email_id AND fans_players.player_id = sales_for_player.player_id
  LEFT JOIN total_sales      ON fans_players.email_id = total_sales.email_id
  LEFT JOIN teradata_exports.lifetime_avidity_scores ON fans_players.email_id = lifetime_avidity_scores.email_id AND fans_players.team_code = lifetime_avidity_scores.team_code
"""

    x_p     = tmp_df[sales_features].values
    y_p     = tmp_df['purchased_for_player']
    mu_p    = np.dot(theta_p.T, x_p.T)
    h_p     = sigmoid(mu_p)
    pt_p    = theta_p[-1]*(h_p - y_p) #/ y_p.shape[0]
    #
    x_v     = tmp_df[votes_features].values
    y_v     = tmp_df['voted_for_player']
    mu_v    = np.dot(theta_v.T, x_v.T)
    h_v     = sigmoid(mu_v)
    pt_v    = theta_v[-1]*(h_v - y_v) #/ y_p.shape[0]
    #
    z_gradient = pt_0 + pt_1 + pt_p + pt_v
    new_z      = y_z + z_gradient * learning_schedule(i)
    tmp_df['old_z'] = tmp_df['z']
    tmp_df['z'] = sigmoid(new_z)
    tmp_df['pt_0'] = pt_0
    tmp_df['pt_1'] = pt_1
    tmp_df['pt_p'] = pt_p
    tmp_df['pt_v'] = pt_v
    tmp_df['z_diff'] = tmp_df['z'] - tmp_df['old_z']
    return tmp_df
